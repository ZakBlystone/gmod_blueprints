LIB LOGIC
{
	SPECIAL If, SHARED
	{
		DESC Conditional 'If' statement
		IN Exec, PN_Exec
		IN Condition, PN_Bool
		OUT True, PN_Exec, #Executes if true
		OUT False, PN_Exec, #Executes if false
		CODE if $2 then #1 end #2
	}
	SPECIAL DoN, SHARED
	{
		DESC Only runs the output N times
		IN Exec, PN_Exec
		IN N=1, PN_Number, #Number of times to run
		IN Reset, PN_Exec, #Reset the internal counter
		OUT Thru, PN_Exec
		OUT Counter, PN_Number, #The current counter value
		EXPERIMENTAL
		GLOBAL counter
		CODE
		{
			if __targetPin == 3 then %counter = $2 goto popcall end
			%counter = %counter or $2
			#2 = %counter
			if %counter > 0 then
			%counter = %counter - 1
			#1
			else
			goto popcall
			end
		}
	}
	PURE And, SHARED
	{
		OUT Result, PN_Bool
		CLASS VariadicOperator
		PARAM operator, and
		REDIRECTPIN A, In_1
		REDIRECTPIN B, In_2
		COMPACT
	}
	PURE Or, SHARED
	{
		OUT Result, PN_Bool
		CLASS VariadicOperator
		PARAM operator, or
		REDIRECTPIN A, In_1
		REDIRECTPIN B, In_2
		COMPACT
	}
	PURE Not, SHARED
	{
		IN A, PN_Bool
		OUT Result, PN_Bool
		CODE #1 = not $1
		COMPACT
	}
	PURE LessThan, SHARED
	{
		DISPLAY <
		IN A, PN_Number
		IN B, PN_Number
		OUT Result, PN_Bool
		CODE #1 = $1 < $2
		COMPACT
	}
	PURE LessThanEqual, SHARED
	{
		DISPLAY <=
		IN A, PN_Number
		IN B, PN_Number
		OUT Result, PN_Bool
		CODE #1 = $1 <= $2
		COMPACT
	}
	PURE GreaterThan, SHARED
	{
		DISPLAY >
		IN A, PN_Number
		IN B, PN_Number
		OUT Result, PN_Bool
		CODE #1 = $1 > $2
		COMPACT
	}
	PURE GreaterThanEqual, SHARED
	{
		DISPLAY >=
		IN A, PN_Number
		IN B, PN_Number
		OUT Result, PN_Bool
		CODE #1 = $1 >= $2
		COMPACT
	}
	PURE Equal, SHARED
	{
		DISPLAY ==
		IN A, PN_Any, PNF_Nullable
		IN B, PN_Any, PNF_Nullable
		OUT Result, PN_Bool
		INFORM 1,2
		CODE #1 = $1 == $2
		COMPACT
	}
	PURE NotEqual, SHARED
	{
		DISPLAY !=
		IN A, PN_Any, PNF_Nullable
		IN B, PN_Any, PNF_Nullable
		OUT Result, PN_Bool
		INFORM 1,2
		CODE #1 = $1 ~= $2
		COMPACT
	}
	PURE Concat, SHARED
	{
		OUT Result, PN_String
		COMPACT
		CLASS VariadicOperator
		REDIRECTPIN A, In_1
		REDIRECTPIN B, In_2
		PARAM operator, ..
	}
	PURE ConcatHuge, SHARED
	{
		DEPRECATED
		IN A, PN_String
		IN B, PN_String
		IN C, PN_String
		IN D, PN_String
		IN E, PN_String
		OUT Result, PN_String
		CODE #1 = $1 .. $2 .. $3 .. $4 .. $5
		COMPACT
	}
}
